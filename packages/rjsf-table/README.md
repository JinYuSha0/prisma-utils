## How to install

```base
npm install rjsf-table
yarn add rjsf-table
```

## How to use

```javascript
import React from "react";
import RJSFForm from "@rjsf/mui";
import { TableField } from "rjsf-table";

const App: React.FC<{}> = () => {
  return (
    <RJSFForm
      formData={yourFormData}
      schema={yourSchema}
      uiSchema={{
        "any-array-items-field": {
          "ui:widget": TableField,
        },
      }}
    />
  );
};
```
